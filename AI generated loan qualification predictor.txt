    <script>
        // Loan Approval Prediction System
        class LoanApprovalPredictor {
            constructor() {
                this.form = document.getElementById('loanForm');
                this.resultContainer = document.getElementById('resultContainer');
                this.closeResultBtn = document.getElementById('closeResult');
                
                this.initializeEventListeners();
            }

            initializeEventListeners() {
                this.form.addEventListener('submit', (e) => this.handleFormSubmit(e));
                this.closeResultBtn.addEventListener('click', () => this.hideResult());
                
                // Add real-time validation
                this.addRealTimeValidation();
            }

            addRealTimeValidation() {
                const inputs = this.form.querySelectorAll('input, select');
                inputs.forEach(input => {
                    input.addEventListener('blur', () => this.validateField(input));
                    input.addEventListener('input', () => this.clearFieldError(input));
                });
            }

            validateField(field) {
                const value = field.value.trim();
                let isValid = true;
                let errorMessage = '';

                switch (field.id) {
                    case 'income':
                        if (value < 0) {
                            isValid = false;
                            errorMessage = 'Income cannot be negative';
                        } else if (value < 100000) {
                            isValid = false;
                            errorMessage = 'Income should be at least R100,000';
                        }
                        break;
                    case 'creditScore':
                        if (value < 300 || value > 850) {
                            isValid = false;
                            errorMessage = 'Credit score must be between 300 and 850';
                        }
                        break;
                    case 'loanAmount':
                        if (value < 1000) {
                            isValid = false;
                            errorMessage = 'Loan amount must be at least R1,000';
                        }
                        break;
                    case 'debtToIncome':
                        if (value < 0 || value > 100) {
                            isValid = false;
                            errorMessage = 'Debt-to-income ratio must be between 0% and 100%';
                        }
                        break;
                    case 'employmentStatus':
                        if (!value) {
                            isValid = false;
                            errorMessage = 'Please select employment status';
                        }
                        break;
                }

                this.showFieldError(field, isValid, errorMessage);
                return isValid;
            }

            showFieldError(field, isValid, message) {
                this.clearFieldError(field);
                
                if (!isValid) {
                    field.style.borderColor = '#dc3545';
                    const errorDiv = document.createElement('div');
                    errorDiv.className = 'field-error';
                    errorDiv.style.color = '#dc3545';
                    errorDiv.style.fontSize = '0.85rem';
                    errorDiv.style.marginTop = '5px';
                    errorDiv.textContent = message;
                    field.parentNode.appendChild(errorDiv);
                }
            }

            clearFieldError(field) {
                field.style.borderColor = '#e1e8ed';
                const errorDiv = field.parentNode.querySelector('.field-error');
                if (errorDiv) {
                    errorDiv.remove();
                }
            }

            async handleFormSubmit(e) {
                e.preventDefault();
                
                // Validate all fields
                const inputs = this.form.querySelectorAll('input, select');
                let allValid = true;
                
                inputs.forEach(input => {
                    if (!this.validateField(input)) {
                        allValid = false;
                    }
                });

                if (!allValid) {
                    return;
                }

                // Show loading state
                const submitBtn = this.form.querySelector('.submit-btn');
                const originalText = submitBtn.innerHTML;
                submitBtn.innerHTML = '<div class="loading"></div> Processing...';
                submitBtn.disabled = true;

                try {
                    // Get form data
                    const formData = new FormData(this.form);
                    const loanData = {
                        income: parseFloat(formData.get('income')),
                        creditScore: parseInt(formData.get('creditScore')),
                        employmentStatus: formData.get('employmentStatus'),
                        loanAmount: parseFloat(formData.get('loanAmount')),
                        debtToIncome: parseFloat(formData.get('debtToIncome'))
                    };

                    // Simulate processing delay
                    await new Promise(resolve => setTimeout(resolve, 1500));

                    // Predict loan approval
                    const prediction = this.predictLoanApproval(loanData);
                    
                    // Display result
                    this.displayResult(prediction, loanData);

                } catch (error) {
                    console.error('Error processing loan prediction:', error);
                    alert('An error occurred while processing your request. Please try again.');
                } finally {
                    // Restore button state
                    submitBtn.innerHTML = originalText;
                    submitBtn.disabled = false;
                }
            }

            predictLoanApproval(loanData) {
                // Calculate risk score based on multiple factors
                let riskScore = 0;
                const factors = [];

                // Income factor (0-25 points)
                const incomeScore = this.calculateIncomeScore(loanData.income);
                riskScore += incomeScore.score;
                factors.push({
                    name: 'Annual Income',
                    impact: incomeScore.impact,
                    details: `R${loanData.income.toLocaleString()} - ${incomeScore.impact}`
                });

                // Credit score factor (0-30 points)
                const creditScore = this.calculateCreditScore(loanData.creditScore);
                riskScore += creditScore.score;
                factors.push({
                    name: 'Credit Score',
                    impact: creditScore.impact,
                    details: `${loanData.creditScore} - ${creditScore.impact}`
                });

                // Employment status factor (0-20 points)
                const employmentScore = this.calculateEmploymentScore(loanData.employmentStatus);
                riskScore += employmentScore.score;
                factors.push({
                    name: 'Employment Status',
                    impact: employmentScore.impact,
                    details: `${loanData.employmentStatus.replace('-', ' ')} - ${employmentScore.impact}`
                });

                // Loan amount factor (0-15 points)
                const loanAmountScore = this.calculateLoanAmountScore(loanData.loanAmount, loanData.income);
                riskScore += loanAmountScore.score;
                factors.push({
                    name: 'Loan Amount',
                    impact: loanAmountScore.impact,
                    details: `R${loanData.loanAmount.toLocaleString()} - ${loanAmountScore.impact}`
                });

                // Debt-to-income ratio factor (0-10 points)
                const dtiScore = this.calculateDTIScore(loanData.debtToIncome);
                riskScore += dtiScore.score;
                factors.push({
                    name: 'Debt-to-Income Ratio',
                    impact: dtiScore.impact,
                    details: `${loanData.debtToIncome}% - ${dtiScore.impact}`
                });

                // Calculate approval probability (0-100)
                const maxScore = 100;
                const approvalProbability = Math.max(0, Math.min(100, ((maxScore - riskScore) / maxScore) * 100));
                
                // Determine approval status
                const isApproved = approvalProbability >= 60;
                const confidence = Math.round(approvalProbability);

                return {
                    approved: isApproved,
                    confidence: confidence,
                    riskScore: riskScore,
                    factors: factors,
                    recommendations: this.generateRecommendations(loanData, factors, isApproved)
                };
            }

            calculateIncomeScore(income) {
                if (income >= 1000000) {
                    return { score: 0, impact: 'Excellent' };
                } else if (income >= 750000) {
                    return { score: 5, impact: 'Good' };
                } else if (income >= 500000) {
                    return { score: 10, impact: 'Fair' };
                } else if (income >= 300000) {
                    return { score: 15, impact: 'Poor' };
                } else {
                    return { score: 25, impact: 'Very Poor' };
                }
            }

            calculateCreditScore(creditScore) {
                if (creditScore >= 750) {
                    return { score: 0, impact: 'Excellent' };
                } else if (creditScore >= 700) {
                    return { score: 5, impact: 'Good' };
                } else if (creditScore >= 650) {
                    return { score: 10, impact: 'Fair' };
                } else if (creditScore >= 600) {
                    return { score: 15, impact: 'Poor' };
                } else {
                    return { score: 30, impact: 'Very Poor' };
                }
            }

            calculateEmploymentScore(employmentStatus) {
                switch (employmentStatus) {
                    case 'employed':
                        return { score: 0, impact: 'Excellent' };
                    case 'self-employed':
                        return { score: 5, impact: 'Good' };
                    case 'retired':
                        return { score: 10, impact: 'Fair' };
                    case 'student':
                        return { score: 15, impact: 'Poor' };
                    case 'unemployed':
                        return { score: 20, impact: 'Very Poor' };
                    default:
                        return { score: 20, impact: 'Very Poor' };
                }
            }

            calculateLoanAmountScore(loanAmount, income) {
                const loanToIncomeRatio = (loanAmount / income) * 100;
                
                if (loanToIncomeRatio <= 20) {
                    return { score: 0, impact: 'Excellent' };
                } else if (loanToIncomeRatio <= 40) {
                    return { score: 5, impact: 'Good' };
                } else if (loanToIncomeRatio <= 60) {
                    return { score: 10, impact: 'Fair' };
                } else {
                    return { score: 15, impact: 'Poor' };
                }
            }

            calculateDTIScore(dtiRatio) {
                if (dtiRatio <= 20) {
                    return { score: 0, impact: 'Excellent' };
                } else if (dtiRatio <= 35) {
                    return { score: 2, impact: 'Good' };
                } else if (dtiRatio <= 45) {
                    return { score: 5, impact: 'Fair' };
                } else if (dtiRatio <= 55) {
                    return { score: 8, impact: 'Poor' };
                } else {
                    return { score: 10, impact: 'Very Poor' };
                }
            }

            generateRecommendations(loanData, factors, isApproved) {
                const recommendations = [];

                if (!isApproved) {
                    // Find the worst factors
                    const poorFactors = factors.filter(f => 
                        f.impact === 'Poor' || f.impact === 'Very Poor'
                    );

                    if (poorFactors.length > 0) {
                        recommendations.push('Focus on improving the following areas:');
                        
                        poorFactors.forEach(factor => {
                            switch (factor.name) {
                                case 'Annual Income':
                                    recommendations.push('Consider increasing your income through additional work or career advancement');
                                    break;
                                case 'Credit Score':
                                    recommendations.push('Work on improving your credit score by paying bills on time and reducing debt');
                                    break;
                                case 'Employment Status':
                                    recommendations.push('Consider finding stable employment to improve your application');
                                    break;
                                case 'Loan Amount':
                                    recommendations.push('Consider requesting a smaller loan amount relative to your income');
                                    break;
                                case 'Debt-to-Income Ratio':

            displayResult(prediction, loanData) {
                // Update prediction badge
                const predictionBadge = document.getElementById('predictionBadge');
                const predictionText = document.getElementById('predictionText');
                const predictionIcon = predictionBadge.querySelector('i');

                if (prediction.approved) {
                    predictionBadge.className = 'prediction-badge approved';
                    predictionText.textContent = 'APPROVED';
                    predictionIcon.className = 'fas fa-check-circle';
                } else {
                    predictionBadge.className = 'prediction-badge rejected';
                    predictionText.textContent = 'REJECTED';
                    predictionIcon.className = 'fas fa-times-circle';
                }

                // Update confidence score
                const confidenceFill = document.getElementById('confidenceFill');
                const confidenceText = document.getElementById('confidenceText');
                
                confidenceFill.style.width = `${prediction.confidence}%`;
                confidenceText.textContent = `${prediction.confidence}%`;

                // Update confidence bar color based on approval
                if (prediction.approved) {
                    confidenceFill.style.background = 'linear-gradient(90deg, #28a745, #20c997)';
                } else {
                    confidenceFill.style.background = 'linear-gradient(90deg, #dc3545, #fd7e14)';
                }

                // Display factors
                this.displayFactors(prediction.factors);

                // Display recommendations
                this.displayRecommendations(prediction.recommendations);

                // Show result container
                this.resultContainer.style.display = 'block';
                
                // Scroll to result
                this.resultContainer.scrollIntoView({ behavior: 'smooth' });
            }

            displayFactors(factors) {
                const factorsList = document.getElementById('factorsList');
                factorsList.innerHTML = '';

                factors.forEach(factor => {
                    const factorItem = document.createElement('div');
                    factorItem.className = 'factor-item';
                    
                    const impactClass = this.getImpactClass(factor.impact);
                    
                    factorItem.innerHTML = `
                        <span class="factor-name">${factor.name}</span>
                        <span class="factor-impact ${impactClass}">${factor.impact}</span>
                    `;
                    
                    factorsList.appendChild(factorItem);
                });
            }

            getImpactClass(impact) {
                switch (impact) {
                    case 'Excellent':
                    case 'Good':
                        return 'positive';
                    case 'Poor':
                    case 'Very Poor':
                        return 'negative';
                    default:
                        return 'neutral';
                }
            }

            displayRecommendations(recommendations) {
                const recommendationsDiv = document.getElementById('recommendations');
                recommendationsDiv.innerHTML = `
                    <h3>Recommendations</h3>
                    <ul>
                        ${recommendations.map(rec => `<li>${rec}</li>`).join('')}
                    </ul>
                `;
            }

            hideResult() {
                this.resultContainer.style.display = 'none';
            }
        }
